182  def __executeCP(self, pathFrom, pathTo):
183  #Setting the process name, and the sequence of arguments
184  allOutput = ""
185  processName = "/usr/bin/scp"
186  command_args = [processName, pathFrom, pathTo]
187 
188  #We now use fork() to create two processes of the same function. However,
189  # the child process will replace itself with the execution of the above commands,
190  # and the parent process will send data to the child process.
191  #PID = 0 for child, and the PID of the child for the parent
192  pid, child_fd = pty.fork()
193  if not pid: # Child process
194  # Replace child process with our SCP process
195  os.execv(processName, command_args)
196 
197  #Back in our parent process, we send the necessary information (like a
198  # password) to our child process
199  while True:
200  output = os.read(child_fd, 1024).decode()
201  if not output:
202  break
203 
204  #Removing return carriages from ouput
205  output = output.replace("\r","")\
206  .replace("\n\n","\n")
207 
208  # Write the password, if the child process is waiting
209  if 'password:' in output.lower():
210  os.write(child_fd, ("{}\n".format(self.config['password'])).encode())
211  #Sleep for just a tiny bit so that our input can be read
212  time.sleep(0.001)
213  elif "are you sure you want to continue connecting" in output.lower():
214  # Adding key to known_hosts
215  os.write(child_fd, ("yes\n").encode())
216  time.sleep(0.001)
217  #Adding the current output to the variable that will hold everything
218  allOutput += output
219  #END WHILE
220  return allOutput
