726  """

729  the TCP threads to complete transfer. Data is a score between 1 and 0, where
730  a time interval is given a 1 if it is transmitting data, and a 0 if it is not. The 1's
731  and 0's are totaled up and averaged.
732  """

736  tcpTests = OBJECT.getTest("TCP", ConnectionLoc=connLoc)

739  #This is just a quick test to see if the TCP tests are in order by number.
740  # If they aren't then we swap them
741  if (len(tcpTests) == 2) and (tcpTests[0].TestNumber > tcpTests[1].TestNumber):
742  #This is a quick swap
743  tcpTests[0], tcpTests[1] = tcpTests[1], tcpTests[0]
744  #END IF
745  _partial_tcpVals = []
746  for test in tcpTests:
747  PERIOD = self.__TCPPeriod(test)
748  RATING = self.__TCPRating(test)

753  tcpVals.extend(_partial_tcpVals)
