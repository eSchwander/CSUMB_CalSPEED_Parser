380  def getTest(self, type_, **kwargs):
381  """
382  Gets the object that meets the specified values. Type of test (PING, TCP, or UDP) must
383  be given, but all other attributes must be given through keyword arguments
384  ARGS:
385  type_: String, the type of test that will be searched for
386  kwargs: Strings, the attributes that the test must have. As keyword
387  arguments, the argument name should be the attribute of the object
388  we are using as criteria. The argument value is the expected value
389  RETURNS:
390  List of type _Test objects that meet the specified attributes in kwargs
391  """
392  #Checking that the type is one of the possible types
393  if type_ not in self.Tests.keys():
394  raise ValueError("The \"type_\" was not of the possible types. "+str(self.Tests.keys()))
395  #If no other attributes were passed in through kwargs, then we just return all
396  # of the test of that type
397  if len(kwargs) == 0:
398  return self.Tests[type_]
399  #Otherwise, we will go through all of the tests, and find the ones that have
400  # the specified attributes.
401  else:
402  matchingTests = []
403  for test in self.Tests[type_]:
404  matchesAll = True
405  #This will go through all of the key/value pairs in kwargs, and test
406  # if the test has the attributes. If any are found to not match, then
407  # the boolean is set to false, and the test will not be added to the
408  # returning array. If the test does not have the attribute at all, then
409  # the boolean is set to false, and the test is not added.
410  for key, value in kwargs.items():
411  try:
412  if test.__dict__[key] != value:
413  matchesAll = False
414  break
415  #This except block will run if the test did not have the attribute "key",
416  # as test.__dict__[key] will cause a KeyError
417  except:
418  matchesAll = False
419  break
420  #END FOR
421  if matchesAll:
422  matchingTests.append(test)
423  return matchingTests
