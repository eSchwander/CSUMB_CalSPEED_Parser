254  def calc_rValMOS(self, delayThreshold=150):
255  """
256  Calculates the R-Value and MOS score for this connection based on the RTT
257  values recorded in the test.
258  ARGS:
259  delayThreshold Integer, the threshold millisecond value of the PING RTT
260  RETURNS:
261  List of two Integers, the R-Value and MOS score of the PING test
262  """
263  if not self.ContainsErrors:
264  #Setting the variables which will hold the float values passed to the
265  # original calc_rVal_MOS function
266  totalCnt = 0.0
267  totalLost = 0.0
268  totalTPng = float(len(self.Times))
269  totalSum = 0.0
270  totalMax = 0.0
271  # F(d) -the rate of packets below delay threshold, done by incrementing
272  # this value for every packet that is below delayThreshold and then dividing by
273  # the total number of measurements. This value is recorded in the variable below
274  totalFd = 0.0
275  #We will now begin to calculate the values that will be passed to the
276  # original calculation function
277  totalMax = self.RTTMax if (totalMax < self.RTTMax) else totalMax
278  totalLost += self.PacketsLost
279  #Now we loop through all of the ping times so that we can add them to the
280  # totalSum variable. If the time retrieved is not 0, the value is added
281  # to the totalSum variable and totalCnt is incremented. We also increment
282  # "Fd" if the time is above the given threshold "delayThreshold"
283  for pingPacket in self.Times:
284  if pingPacket.RTT > 0:
285  totalCnt += 1
286  totalSum += pingPacket.RTT
287  if (pingPacket.RTT < delayThreshold):
288  totalFd+=1
289  #END IF
290  #END IF
291  #END FOR
292  #Returning (rVal, MOS)
293  if totalCnt != 0:
294  return calc_rVal_MOS(totalSum, totalCnt, totalTPng, totalLost, totalFd)
295  else:
296  return [0.0,1.0]
297  else:
298  return [self.ErrorType]*2
