246  def __getAllCoordinates(self, fileStream):
247  """
248  A function used by loadFieldTestInfo to parse all of the Location pairs
249  (i.e. Latitude and Longitude), and save in self.AllCoordPairs
250  """
251  #Getting all lines with the Latitude info
252  latitudes = getLinesWith(fileStream, "Latitude:")
253  #Now that we have all of the latitudes, we need to go back to those locations
254  # in the file and get the following lines, which contain the Longitudes
255  longitudes = []
256  oldLoc = fileStream.tell()
257  for goodLat in latitudes:
258  line = "__"
259  while line:
260  if goodLat in line:
261  line = fileStream.readline()
262  longitudes.append(line)
263  break
264  line = fileStream.readline()
265  #END WHILE
266  if latitudes.index(goodLat) != (len(longitudes)-1):
267  longitudes.append("Longitude: 0.0")
268  #END FOR
269  fileStream.seek(oldLoc)
270 
271  pairs = []
272  #Now we are going to go through our three arrays, grabbing each value
273  # and making a 2-element array out of the latitude and longitude.
274  #We will only go along the array up to the shortest array so that we don't try to
275  # index beyond the array
276  for i in range( len(latitudes) ):
277  pairs.append( [latitudes[i].split("Latitude:")[1].strip(),
278  longitudes[i].split("Longitude:")[1].strip()] )
279  #END FOR
280  #Going through each 'pair' (of three elements) and casting them to a float if they
281  # are numeric. Otherwise, we assume that they are 0
282  for pair in pairs:
283  newPair = []
284  for elem in pair:
285  try:
286  elem = float(elem)
287  newPair.append(elem)
288  except:
289  newPair.append(0)
290  #END FOR
291  pairs[pairs.index(pair)] = newPair
292  #END FOR
293  #Making all of our 3-element pairs into tuples
294  pairs = [tuple(pair) for pair in pairs]
295  return pairs

298  def findAndParseUDPTests(self):
299  """
300  This takes the contents of the file being parsed, splits the content by "Staring Test"
301  (to included any error messages in the tests) using the readAllTestsFromFile function,
302  and the creates the specified Test objects. Assumes that self._fileContentsByTest
303  contains all of the tests.
304  """
305  #First we run the readAllTestsFromFile function, to make sure that self._fileContentsByTest
306  # is set, and contains all of the test output
307  self.readAllTestsFromFile()
308  #If the function above ran and did not hit any major errors, then we can run the code
309  # inside of the IF block
310  if not self.ContainsErrors:
311  for chunk in self._fileContentsByTest:
312  if "1 second Test" in chunk:
313  #This block gets the test number from the beginning of this chunk of
314  # characters. Each 1 second test has a different test number, as 3 are
315  # run in sequence. The number stored in tempTestNum applies to all 3
316  allStartingLine = ""
317  tempTestNum = 0
318  for line in chunk.split("\n"):
319  #If "Starting Test" is in the line, and we have not set our overall
320  # test number yet, we are going to parse it out. We also save the
321  # line with the number in it so that we can concatenate it to
322  # the beginning of each chunk
323  if ("Starting Test" in line) and (tempTestNum == 0):
324  allStartingLine = line+"\n"
325  rightChunk = line.split("Starting Test ")[1].strip()
326  tempTestNum = rightChunk.split(":")[0].split("..")[0]
327  break
328  #END FOR
329  subTests = chunk.split("Starting UDP 1")[1:]
330  subTests = [(allStartingLine+"Starting UDP 1"+text) for text in subTests]
331  #Now we go through each 1 second test, parse it into a UDP_Test object,
332  # and appended it to the array parsedSubTests. This array will then be
333  # added to self.Tests in the UDP and NUM category
334  for test in subTests:
335  #The if statement is one last check to make sure that the test actually contains
336  # some basic information
337  if "Iperf command line" in test:
338  parsedTest = UDP_Test(dataString=test, eastWestIP=self.EastWestSrvrIPs)
339  tempSubTestNum = test.split("Test #")[1].split("\n")[0].strip()
340  parsedTest.TestNumber = int(tempTestNum)
341  parsedTest.SubTestNumber = int(tempSubTestNum)
342  self.Tests[parsedTest.ConnectionType].append(parsedTest)
343  #Determining the index for this "sub" test
344  byNum = float(str(parsedTest.TestNumber)+"."+str(parsedTest.SubTestNumber))
345  self.TestsByNum[byNum] = parsedTest
346  #END IF
347  #END FOR
348  else:
349  parsedTest = UDP_Test(dataString=chunk, eastWestIP=self.EastWestSrvrIPs)
350  #If the line above returned an object (and not None), then we have correctly
351  # parsed a Test, and can add it to our list.
352  if parsedTest:
353  self.Tests[parsedTest.ConnectionType].append(parsedTest)
354  self.TestsByNum[int(parsedTest.TestNumber)] = parsedTest
355  else:
356  pass #If we are passing, then the test was not a TCP test
357  #END IF/ELSE
358  #END IF/ELSE
359  #END FOR
360  #END IF
