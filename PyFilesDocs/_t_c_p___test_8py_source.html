291  def get_ThreadSumValues(self, direction="DOWN", attribute="Speed"):
292  """
293  Creating an array of the sum of each 1 second interval of all 4 thread's speed or size
294  ARGS:
295  direction String, threads of specified direction (Up or Down) that will be summed
296  attribute String, the attribute of the measurement we wish to sum up (speed or size)
297  RETURNS:
298  threads_summed List, containing values representing the sum of each 4 threads' speed
299  """
300  #Just in case what was passed is not a possible option, we will chose the defaults
301  if attribute not in ["Speed", "Size"]:
302  attribute = "Speed"
303  if direction not in self.Threads.keys():
304  direction = "DOWN"
305  #END IF
306  #Getting all of the necessary values from the TCP Thread objects we have
307  allValues = self.get_ThreadsValues(direction, attribute)
308  #Calculating max thread length
309  maxThreadLength = max( [len(array) for array in allValues] )
310  #Get the sums of the "direction" threads. The first FOR loop will iterate X number of times
311  # based on what is in maxThreadLength. Within the FOR loop, we set up
312  # a temporary variable. We then try to add a specific interval (e.g. 1.0-2.0 sec) for each
313  # thread to the temporary variable. If the interval does not exist, then nothing is added
314  # (i.e. we add 0). We then append this value to threads_summed, which will be returned.
315  threads_summed = []
316  for interval in range(maxThreadLength):
317  temp = 0
318  for array in allValues:
319  try:
320  temp += array[interval]
321  except:
322  pass
323  #END FOR
324  threads_summed.append(temp)
325  #END FOR
326  return threads_summed
