405  def __getLocationCoords(self, fileStream, type_=""):
406  """
407  A function used by loadCrowdSourceInfo to parse for multiple location coordinates
408  in the file based on type (everything, GPS, or Network)
409  """
410  if type_ not in ["", "GPS", "Network"]:
411  raise ValueError("Your type of Location values were not one of the options:"+str(["", "GPS", "Network"]))
412  #Now we will look for the specified types of Location information
413  latitudes = getLinesWith(fileStream, type_+"Latitude:")
414  longitudes = getLinesWith(fileStream, type_+"Longitude:")
415  distsMoved = getLinesWith(fileStream, type_+"DistanceMoved:")
416 
417  pairs = []
418  #Now we are going to go through our three arrays, grabbing each value
419  # and making a 3-element array out of the latitude, longitude, and distance moved
420  #We will only go along the array up to the shortest array so that we don't try to
421  # index beyond the array
422  for i in range( min([len(latitudes),len(longitudes),len(distsMoved)]) ):
423  pairs.append( [latitudes[i].split(type_+"Latitude:")[1].strip(),
424  longitudes[i].split(type_+"Longitude:")[1].strip(),
425  distsMoved[i].split(type_+"DistanceMoved:")[1].strip()] )
426  #END FOR
427  #Going through each 'pair' (of three elements) and casting them to a float if they
428  # are numeric. Otherwise, we assume that they are 0
429  for pair in pairs:
430  newPair = []
431  for elem in pair:
432  try:
433  elem = float(elem)
434  newPair.append(elem)
435  except:
436  newPair.append(0)
437  #END FOR
438  pairs[pairs.index(pair)] = newPair
439  #END FOR
440  #Making all of our 3-element pairs into tuples
441  pairs = [tuple(pair) for pair in pairs]
442  return pairs
