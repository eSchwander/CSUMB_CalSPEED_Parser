31  fileStream: FileStream object, called with open(FILEPATH, 'r')

89  return (None, None, [])

105 def calcTCPThroughput(RTT, MSS=1024, Loss=0.000001):
106  """
107  Given a few values (RTT is required), this function will return
108  the theoretical TCP Throughput on the connection from which the RTT was obtained.
109  RTT must be in milliseconds
110  MSS must be in bytes
111  Loss must be in percent
112  Returns the theoretical throughput in bits/sec.
113  ARGS:
114  RTT Integer/Float, the round trip time it took to send a packet on a connection.
115  Cannot be a string, None, or 0
116  MSS Integer/Float, default 1024, the maximum segment size that can be sent in the TCP connection.
117  Cannot be a string, None, or 0
118  Loss Integer/Float, default 0.01, the percentage of packets that will be lost in the connection.
119  Cannot be a string, None, or 0
120  RETURNS:
121  Integer/None Returns None if wrong type of value was given, otherwise an Integer
122  """
123  for value in [RTT, MSS, Loss]:
124  if (value is None) or (isinstance(value, str)):
125  return None
126  #END FOR
127  for value in [RTT, MSS, Loss]:
128  if (value == 0):
129  return 0
130  #END FOR
131  RTT_calc = RTT / 1000.0
132  MSS_calc = MSS * 8.0 / 1024.0
133  Loss_calc = Loss / 100.0
134  from math import sqrt
135  return ( (MSS_calc / RTT_calc) / sqrt(Loss_calc) )

138 def calc_rVal_MOS(pktSum, pktCount, pktExpected, pktLost, pktFd):
139  """
140  This takes a few given numbers that come from determining RTT and packet loss,
141  and returns the rVal and MOS as a tuple.
142  ARGS:
143  pktSum Float, sum of RTT values of all pings in a Ping test
144  pktCount Float, number of pings in Ping test that did not fail
145  pktExpected Float, number of successful pings expected
146  pktLost Float, number of unsuccessful pings in a Ping test
147  pktFd Float, number of pings whose RTT was less than a given limit (150 ms)
148  RETURNS:
149  (rVal, MOS) Tuple of Floats
150  """
151  #Doing some value checking
152  for elem in [pktSum, pktCount, pktExpected, pktLost, pktFd]:
153  if not isinstance(elem, float):
154  raise TypeError
155  #END FOR
156  #Calculating the average RTT (Sum is a sum of all RTT, Count is how many)
157  pktAverage = pktSum / pktCount
158 
159  #pktFd is the number of packets (i.e. pings) where the RTT was below
160  # a certain threshold. We divide that by how many pings were sent (pktExpected)
161  # to get a decimal representation of the ratio.
162  pktFd = pktFd / pktExpected
163 
164  #Loss rate is calculate using this formula
165  P_n = pktLost / pktCount
166  #Loss rate of jitter buffer is calculated using this formula
167  P_b = (1 - P_n) * (1 - pktFd)
168  #Equipment Impairment Factor
169  I_eEff = 5 + (90 * (P_n+P_b) / (P_n+P_b+10) )
170  #Calculating H(x). This is either 0 or 1
171  H_of_X = 0 if ((pktAverage-177.3) < 0) else 1
172  #Calculating Delay Impairment Factor
173  I_d = (0.024 * pktAverage) + (0.11 * (pktAverage-177.3) * H_of_X)
174 
175  #Now we can finally calculate the rValue and the MOS
176  rValue = 93.2 - I_d - I_eEff
177  MOS = 1 if (rValue <= 0) else ( 1 + (0.035*rValue) + (rValue * (rValue-60) * (100-rValue) * 7 * 0.000001 ) )
178  #This line is for rounding off the extra numbers on the end of the MOS value.
179  # We just want 1 place after the decimal
180  MOS = float(round(MOS*10))/10.0
181  #And now we return a tuple of the rVal and MOS value. Returned in a tuple
182  # because I didn't want to use an array. Deal with it
183  return [rValue, MOS]
